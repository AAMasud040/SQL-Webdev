    title VARCHAR(30) NOT NULL,
    description VARCHAR(1000),
    date DATE DEFAULT '2021-01-01',
    lang_id int,
    duration Double,
    
    CONSTRAINT film_tab_pk1 PRIMARY KEY(id),
    CONSTRAINT film_tab_uqk1 UNIQUE KEY(title),

2) CREATE TABLE IF NOT EXISTS PC(
    id VARCHAR(5),
    model_name VARCHAR(30) NOT NULL,
    release_date DATE,
    country_of_origin VARCHAR(40) DEFAULT 'Bangladesh',
    warranty LONGTEXT,
    
    CONSTRAINT pc_table_uqk_1 UNIQUE(release_date),
    CONSTRAINT pc_table_pk_1 PRIMARY KEY(id)
    );

3) CREATE TABLE IF NOT EXISTS category(
    id INT,
    name VARCHAR(100) NOT NULL,
    
    CONSTRAINT category_table_pk1 PRIMARY KEY(id),
    CONSTRAINT category_table_uqk1 UNIQUE(name)
    );

DemoData:
	INSERT INTO category VALUES(1,'Keyboard'),(2,'Mouse');

4) CREATE TABLE IF NOT EXISTS component(
	id VARCHAR(10),
    name VARCHAR(30),
    category int,
    configuration VARCHAR(300),
    price DOUBLE(10,2) NOT NULL,
    
    CONSTRAINT component_table_pk1 PRIMARY KEY(id),
    CONSTRAINT Fk2 FOREIGN KEY(category) REFERENCES category(id)
    );

5) CREATE TABLE IF NOT EXISTS parts(

    type VARCHAR(20) NOT NULL,
    
    CONSTRAINT parts_table_pk1 PRIMARY KEY(id),
    CONSTRAINT parts_table_uqk1 UNIQUE KEY(type)

 CREATE TABLE IF NOT EXISTS pc_parts(
    	pc_id VARCHAR(5),
    	comp_id VARCHAR(10),
    	part_type VARCHAR(20) NOT NULL,
    
    	CONSTRAINT pc_parts_pk1 PRIMARY KEY(pc_id,comp_id),
